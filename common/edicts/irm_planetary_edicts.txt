# Relocate sector capital
planet_edict = {
	name = "sector_capital"
	cost = { influence = @standardEdictCost }
	length = 0
	potential = {
		sector_controlled = yes
		sector = { any_planet = { is_sector_capital = yes } }
	}
	allow = {
		custom_tooltip = {
			fail_text = "edict_sector_capital.status"
			not = { has_modifier = "sector_capital" }
		}			
	}
	effect = {
		custom_tooltip = "edict_sector_capital.effect"
		hidden_effect = { planet_set_sector_capital = yes }
	}
	ai_weight = { weight = 0 }
}
