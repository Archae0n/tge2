# Buttons for Sector Control screen

# UI / DROPDOWN

# Opens additional sector overview
ui_menu_sector_control = {
	potential = { always = yes }
	allow = {
		custom_tooltip = {
			fail_text = "sector_control.ui.disallow.nosector"
			and = {
				sectors > 0
				any_sector = {
					sector_has_db = yes
					any_planet = { is_sector_capital = yes }
				}
			}
		}
		custom_tooltip = {
			fail_text = "sector_control.ui.disallow.duplicated"
			not = { has_country_flag = "ui_sector_control_active" }
		}		
	}
	effect = {
		custom_tooltip = "sector_control.ui.effect"
		hidden_effect = {
			country_event = { id = irm_sector.1 }
		}
	}	
}
