effect_play_as_ruler_set_target_on_them = {
	random_owned_leader = {
		limit = {
			leader_class = ruler
			has_leader_flag = play_as_ruler_ruler_itself
		}
		save_event_target_as = ruler
	}
}

effect_play_as_ruler_hobby_event_ended = {
	random_owned_leader = {
		limit = {
			leader_class = ruler
			has_leader_flag = play_as_ruler_ruler_itself
		}
		save_event_target_as = ruler
	}
	event_target:ruler = {
		set_timed_leader_flag = { flag = play_as_ruler_hobby_event_cooldown days = 720 }
	}
}
