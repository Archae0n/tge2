## Lost Colony Triggers
#scope: country
is_cbshing_lost_colony_country = {
	OR = {
		has_valid_civic = civic_cbshing_lost_colony
		has_country_flag = human_2
	}
}
#scope: planet
is_cbshing_lost_colony_home_planet = {
	has_planet_flag = lost_colony_homeworld
}
#scope: country
is_valid_cbshing_lost_colony_home_country = {
	NOR = { 
		has_valid_civic = civic_cbshing_lost_colony
		has_country_flag = human_2
		has_valid_civic = civic_cbshing_nuclear_mutation
		has_valid_civic = civic_cbshing_voidborne
		has_valid_civic = civic_post_apocalyptic
		has_valid_civic = civic_life_seeded 
		has_ethic = ethic_gestalt_consciousness 
	}
}

has_cbshing_nuclear_mutation_secondary_species = {
	OR = {
		has_valid_civic = civic_cbshing_nuclear_vault
		has_valid_civic = civic_cbshing_robo_nuked_start
		has_valid_civic = civic_cbshing_nuclear_mutation
	}
	OR = {
		has_valid_civic = civic_syncretic_evolution
		AND = {
			has_valid_civic = civic_cbshing_robo_nuked_start
			OR = {
				has_government = gov_machine_servitor
				has_government = gov_machine_assimilator
			}
		}
	}
}
