namespace = gai_ai_armies
country_event = {
	id = gai_ai_armies.1	
	hide_window = yes
	#is_triggered_only = yes
	
	mean_time_to_happen = { 
		months = 3 
	}
	
	immediate = {
		IF = {
			limit = {
				AND = {
					is_ai = yes
					count_armies = {
						limit = { army_type = assault_army }
						count < 8
					}
					is_country_type = "default"
					OR = {
						has_monthly_income = {
							resource = energy
							value > 50
						}
						is_at_war = yes
					}
					OR = {
						minerals > 1250
						AND = {
							minerals > 100
							is_at_war = yes
						}
					}
					NOT = { has_global_flag = disable_gai_ai_armies }
					any_owned_planet = {
						is_occupied_flag = no
						has_orbital_bombardment = no
					}
				}
			}
			random_owned_planet = {
				limit = {
					is_occupied_flag = no
					has_orbital_bombardment = no
				}
				create_army = {
					owner = from
					type = "assault_army"
				}
			}
			add_minerals = -100
		}
	}
}
