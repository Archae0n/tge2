			# Blob effect is used to create some kind of sorting, based on distance.
			effect_on_blob = {
				center = capital_scope.solar_system

					}
				}
				effect = {
					# +1 to pointer
					owner = { change_variable = { which = nSectorPointer value = 1 } }
					# Choose capital for each sector with hard criterias to avoid mess.
					random_system_planet = {
						limit = {
							has_modifier = "sector_capital"
							exists = sector
							sector = {
								is_core_sector = no
								sector_has_db = yes
								prev = { has_planet_flag = sector_id_@prev }
							}
						}
						# Using pointer to assign specific temporary target to selected capital
						set_variable = { which = nSectorPointer value = owner }
						save_sector_slots = yes
						set_variable = { which = nSectorPointer value = 0 }
						# Collect data to database scopes
						sector = {
							sector_estimate_effects			= yes
							sector_estimate_system_cap 		= yes
							sector_estimate_system_num 		= yes
							sector_estimates_fleet_cap		= yes
							sector_estimate_regionalism		= yes
							#sector_ui_estimate_naval_size   = yes
						}
					}
				}
			}
			set_variable = { which = nSectorPointer value = 0 }
		}
	}
