namespace = psc
# Tradition Discovery - Exploration Journals gain research points.
ship_event = {
	id = psc.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				owner = {
					OR = {
						has_non_swapped_tradition = tr_plentiful_experimentalism_4
						has_swapped_tradition = tr_plentiful_experimentalism_4
					}
				}
			}
			owner = {
				add_monthly_resource_mult = {
					resource = physics_research
					value = 0.1
					min = 1
				}
				add_monthly_resource_mult = {
					resource = society_research
					value = 0.1
					min = 1
				}
				add_monthly_resource_mult = {
					resource = engineering_research
					value = 0.1
					min = 1
				}
			}
		}
	}
}
